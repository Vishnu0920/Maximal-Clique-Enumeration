Build and Run Instructions

Prerequisites

  • C++ compiler with C++17 support (GCC 7+ or Clang 5+)
  • CMake 3.10 or higher
  • Git (for cloning the repository)
  • Python 3.6+ (for running analysis scripts)

Clone the Repository


git clone https://github.com/yourusername/maximal-clique-enumeration.git
cd maximal-clique-enumeration

Build the Project


mkdir build
cd build
cmake ..
make

Run the Algorithms


# Run all algorithms on a specific dataset
./maximal_clique ../datasets/karate.txt

# Run a specific algorithm on a dataset
./maximal_clique ../datasets/karate.txt --algorithm clique
./maximal_clique ../datasets/karate.txt --algorithm bron_kerbosch
./maximal_clique ../datasets/karate.txt --algorithm arboricity

# Run with timing information
./maximal_clique ../datasets/karate.txt --time

# Run all experiments and save results
../scripts/run_experiments.sh

Generate Synthetic Graphs


python ../scripts/generate_graphs.py --type erdos_renyi --n 500 --p 0.1 --output ../datasets/random_500_01.txt
python ../scripts/generate_graphs.py --type barabasi_albert --n 1000 --m 3 --output ../datasets/barabasi_1000.txt
python ../scripts/generate_graphs.py --type watts_strogatz --n 1000 --k 10 --p 0.1 --output ../datasets/watts_strogatz_1000.txt

Plot Results


python ../scripts/plot_results.py --input ../results/runtime_results.csv --output runtime_plot.png

void CLIQUE::findMaximalCliques(const Graph& G, std::vector>& result) {
    std::vector R, P, X;
    
    // Initialize P with all vertices
    for (int i = 0; i < G.getNumVertices(); i++) {
        P.push_back(i);
    }
    
    // Start the recursive algorithm
    findMaximalCliquesRecursive(G, R, P, X, result);
}

void CLIQUE::findMaximalCliquesRecursive(
    const Graph& G, 
    std::vector& R, 
    std::vector& P, 
    std::vector& X, 
    std::vector>& result
) {
    // If P and X are both empty, R is a maximal clique
    if (P.empty() && X.empty()) {
        result.push_back(R);
        return;
    }
    
    // Choose a pivot vertex u from P ∪ X
    int u = choosePivot(G, P, X);
    
    // Get neighbors of u
    std::vector neighbors = G.getNeighbors(u);
    std::vector isNeighbor(G.getNumVertices(), false);
    for (int v : neighbors) {
        isNeighbor[v] = true;
    }
    
    // Make a copy of P to iterate through
    std::vector P_copy = P;
    
    // For each vertex v in P \ N(u)
    for (int v : P_copy) {
        if (isNeighbor[v]) continue;
        
        // Add v to R
        R.push_back(v);
        
        // Create P ∩ N(v) and X ∩ N(v)
        std::vector P_new, X_new;
        std::vector v_neighbors = G.getNeighbors(v);
        std::vector isVNeighbor(G.getNumVertices(), false);
        for (int w : v_neighbors) {
            isVNeighbor[w] = true;
        }
        
        for (int w : P) {
            if (isVNeighbor[w]) {
                P_new.push_back(w);
            }
        }
        
        for (int w : X) {
            if (isVNeighbor[w]) {
                X_new.push_back(w);
            }
        }
        
        // Recursive call
        findMaximalCliquesRecursive(G, R, P_new, X_new, result);
        
        // Remove v from R
        R.pop_back();
        
        // Move v from P to X
        P.erase(std::remove(P.begin(), P.end(), v), P.end());
        X.push_back(v);
    }
}

int CLIQUE::choosePivot(const Graph& G, const std::vector& P, const std::vector& X) {
    // Choose the vertex with the most neighbors in P
    int max_connections = -1;
    int pivot = -1;
    
    // Check vertices in P
    for (int u : P) {
        int connections = 0;
        std::vector neighbors = G.getNeighbors(u);
        for (int v : neighbors) {
            if (std::find(P.begin(), P.end(), v) != P.end()) {
                connections++;
            }
        }
        if (connections > max_connections) {
            max_connections = connections;
            pivot = u;
        }
    }
    
    // Check vertices in X
    for (int u : X) {
        int connections = 0;
        std::vector neighbors = G.getNeighbors(u);
        for (int v : neighbors) {
            if (std::find(P.begin(), P.end(), v) != P.end()) {
                connections++;
            }
        }
        if (connections > max_connections) {
            max_connections = connections;
            pivot = u;
        }
    }
    
    return pivot;
}

void BronKerbosch::findMaximalCliques(const Graph& G, std::vector>& result) {
    // Compute degeneracy ordering
    std::vector ordering = computeDegeneracyOrdering(G);
    
    // For each vertex in the ordering
    for (size_t i = 0; i < ordering.size(); i++) {
        int v = ordering[i];
        
        // Initialize R, P, and X
        std::vector R = {v};
        std::vector P, X;
        
        // Get neighbors of v
        std::vector neighbors = G.getNeighbors(v);
        
        // Add neighbors that come later in the ordering to P
        for (int u : neighbors) {
            auto it = std::find(ordering.begin() + i + 1, ordering.end(), u);
            if (it != ordering.end()) {
                P.push_back(u);
            }
        }
        
        // Add neighbors that come earlier in the ordering to X
        for (int u : neighbors) {
            auto it = std::find(ordering.begin(), ordering.begin() + i, u);
            if (it != ordering.begin() + i) {
                X.push_back(u);
            }
        }
        
        // Call the recursive function
        findMaximalCliquesRecursive(G, R, P, X, result);
    }
}

void BronKerbosch::findMaximalCliquesRecursive(
    const Graph& G, 
    std::vector& R,

        // Remove v from R
        R.pop_back();

        P.erase(std::remove(P.begin(), P.end(), v), P.end());
        X.push_back(v);
    }
}

std::vector BronKerbosch::computeDegeneracyOrdering(const Graph& G) {
    int n = G.getNumVertices();
    std::vector ordering(n);
    std::vector removed(n, false);
    std::vector degrees(n);
    
    // Initialize degrees
    for (int i = 0; i < n; i++) {
        degrees[i] = G.getNeighbors(i).size();
    }
    
    // Compute ordering
    for (int i = 0; i < n; i++) {
        // Find the vertex with minimum degree
        int min_degree = INT_MAX;
        int min_vertex = -1;
        for (int v = 0; v < n; v++) {
            if (!removed[v] && degrees[v] < min_degree) {
                min_degree = degrees[v];
                min_vertex = v;
            }
        }
        
        // Add to ordering and mark as removed
        ordering[i] = min_vertex;
        removed[min_vertex] = true;
        
        // Update degrees of neighbors
        for (int u : G.getNeighbors(min_vertex)) {
            if (!removed[u]) {
                degrees[u]--;
            }
        }
    }
    
    return ordering;
}

int BronKerbosch::choosePivot(const Graph& G, const std::vector& P, const std::vector& X) {
    // Same as CLIQUE algorithm
    int max_connections = -1;
    int pivot = -1;
    
    // Check vertices in P
    for (int u : P) {
        int connections = 0;
        std::vector neighbors = G.getNeighbors(u);
        for (int v : neighbors) {
            if (std::find(P.begin(), P.end(), v) != P.end()) {
                connections++;
            }
        }
        if (connections > max_connections) {
            max_connections = connections;
            pivot = u;
        }
    }
    
    // Check vertices in X
    for (int u : X) {
        int connections = 0;
        std::vector neighbors = G.getNeighbors(u);
        for (int v : neighbors) {
            if (std::find(P.begin(), P.end(), v) != P.end()) {
                connections++;
            }
        }
        if (connections > max_connections) {
            max_connections = connections;
            pivot = u;
        }
    }
    
    return pivot;
}

Arboricity-based Algorithm Implementation


#include "arboricity.h"
#include 
#include 

void ArboricityClique::findMaximalCliques(const Graph& G, std::vector>& result) {
    // Compute an acyclic orientation with bounded outdegree
    std::vector> outNeighbors = computeAcyclicOrientation(G);
    
    // For each vertex
    for (int v = 0; v < G.getNumVertices(); v++) {
        // Initialize C and P
        std::vector C = {v};
        std::vector P = outNeighbors[v];
        
        // Call the recursive function
        findMaximalCliquesRecursive(G, outNeighbors, C, P, result);
    }
}

void ArboricityClique::findMaximalCliquesRecursive(
    const Graph& G,
    const std::vector>& outNeighbors,
    std::vector& C,
    std::vector& P,
    std::vector>& result
) {
    // If P is empty, C is a maximal clique
    if (P.empty()) {
        result.push_back(C);
        return;
    }
    
    // Make a copy of P to iterate through
    std::vector P_copy = P;
    
    // For each vertex u in P
    for (int u : P_copy) {
        // Add u to C
        C.push_back(u);
        
        // Create P ∩ N+(u)
        std::vector P_new;
        for (int w : outNeighbors[u]) {
            if (std::find(P.begin(), P.end(), w) != P.end()) {
                P_new.push_back(w);
            }
        }
        
        // Recursive call
        findMaximalCliquesRecursive(G, outNeighbors, C, P_new, result);
        
        // Remove u from C
        C.pop_back();
        
        // Remove u from P
        P.erase(std::remove(P.begin(), P.end(), u), P.end());
    }
}

std::vector> ArboricityClique::computeAcyclicOrientation(const Graph& G) {
    int n = G.getNumVertices();
    std::vector> outNeighbors(n);
    
    // Compute degeneracy ordering (similar to Bron-Kerbosch)
    std::vector ordering = computeDegeneracyOrdering(G);
    
    // Create a map from vertex to its position in the ordering
    std::vector position(n);
    for (int i = 0; i < n; i++) {
        position[ordering[i]] = i;
    }
    
    // Orient edges from lower to higher position in the ordering
    for (int i = 0; i < n; i++) {
        int v = ordering[i];
        for (int u : G.getNeighbors(v)) {
            if (position[u] > position[v]) {
                outNeighbors[v].push_back(u);
            }
        }
    }
    
    return outNeighbors;
}

    int n = G.getNumVertices();
    std::vector ordering(n);
    std::vector removed(n, false);
    std::vector degrees(n);
    
    // Initialize degrees
    for (int i = 0; i < n; i++) {
        degrees[i] = G.getNeighbors(i).size();
    }
    
    // Compute ordering
    for (int i = 0; i < n; i++) {
        // Find the vertex with minimum degree
        int min_degree = INT_MAX;

        }
    }
    
    return ordering;
}

Graph Data Structure Implementation


#include "graph.h"
#include 
#include 
#include 

Graph::Graph(int n) : numVertices(n) {
    adjacencyList.resize(n);
}

Graph::Graph(const std::string& filename) {
    loadFromFile(filename);
}

void Graph::addEdge(int u, int v) {
    // Ensure vertices are valid
    if (u >= numVertices || v >= numVertices || u < 0 || v < 0) {
        std::cerr << "Invalid vertex indices: " << u << ", " << v << std::endl;
        return;
    }
    
    // Add edge (undirected graph, so add both directions)
    adjacencyList[u].push_back(v);
    adjacencyList[v].push_back(u);
}

std::vector Graph::getNeighbors(int v) const {
    if (v >= numVertices || v < 0) {
        std::cerr << "Invalid vertex index: " << v << std::endl;
        return std::vector();
    }
    return adjacencyList[v];
}

int Graph::getNumVertices() const {
    return numVertices;
}

int Graph::getNumEdges() const {
    int count = 0;
    for (const auto& neighbors : adjacencyList) {
        count += neighbors.size();
    }
    // Each edge is counted twice (once for each endpoint)
    return count / 2;
}

double Graph::getDensity() const {
    int n = getNumVertices();
    int m = getNumEdges();
    // Density = 2m / (n(n-1))
    return (n <= 1) ? 0.0 : (2.0 * m) / (n * (n - 1));
}

int Graph::getMaxDegree() const {
    int maxDegree = 0;
    for (const auto& neighbors : adjacencyList) {
        maxDegree = std::max(maxDegree, static_cast(neighbors.size()));
    }
    return maxDegree;
}

void Graph::loadFromFile(const std::string& filename) {
    std::ifstream file(filename);
    if (!file.is_open()) {
        std::cerr << "Error opening file: " << filename << std::endl;
        return;
    }
    
    // First pass: determine the number of vertices
    int maxVertex = -1;
    std::string line;
    while (std::getline(file, line)) {
        // Skip comment lines
        if (line.empty() || line[0] == '#') continue;
        
        std::istringstream iss(line);
        int u, v;
        if (iss >> u >> v) {
            maxVertex = std::max(maxVertex, std::max(u, v));
        }
    }
    
    // Set the number of vertices (0-indexed)
    numVertices = maxVertex + 1;
    adjacencyList.resize(numVertices);
    
    // Second pass: add edges
    file.clear();
    file.seekg(0);
    while (std::getline(file, line)) {
        // Skip comment lines
        if (line.empty() || line[0] == '#') continue;
        
        std::istringstream iss(line);
        int u, v;
        if (iss >> u >> v) {
            addEdge(u, v);
        }
    }
    
    file.close();
}

void Graph::printInfo() const {
    std::cout << "Graph Information:" << std::endl;
    std::cout << "  Vertices: " << getNumVertices() << std::endl;
    std::cout << "  Edges: " << getNumEdges() << std::endl;
    std::cout << "  Density: " << getDensity() << std::endl;
    std::cout << "  Maximum Degree: " << getMaxDegree() << std::endl;
}
