// CLIQUE Algorithm Implementation
function clique(R, P, X) {
    if (P.size === 0 && X.size === 0) {
        reportClique(R);
        return;
    }
    
    for (const v of P) {
        const neighbors = getNeighbors(v);
        clique(new Set([...R, v]), 
               intersection(P, neighbors),
               intersection(X, neighbors));
        P.delete(v);
        X.add(v);
    }
}
